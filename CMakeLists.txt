cmake_minimum_required( VERSION 3.16 )

project( Sigma VERSION 1.0.5 LANGUAGES CXX )

set( CMAKE_CXX_STANDARD 17 )

set( CMAKE_CXX_STANDARD_REQUIRED ON )

find_package( Qt6 REQUIRED COMPONENTS Quick Graphs )

if ( WIN32 )
#! [appicon_windows]
    set( app_icon_resource_windows
        "${CMAKE_CURRENT_SOURCE_DIR}/resources/icons/icon.rc"
    )
    qt_add_executable( Sigma src/main.cpp ${app_icon_resource_windows} )
#! [appicon_windows]
elseif ( APPLE )
#! [appicon_macOS]
    # The MACOSX_BUNDLE_ICON_FILE variable is added to the Info.plist
    # generated by CMake. This variable contains the .icns file name,
    # without the path.
    set( MACOSX_BUNDLE_ICON_FILE sigma.icns )

    # And the following tells CMake where to find and install the file itself.
    set( app_icon_macos
        "${CMAKE_CURRENT_SOURCE_DIR}/resources/icons/sigma.icns"
    )
    set_source_files_properties( ${app_icon_macos} PROPERTIES
        MACOSX_PACKAGE_LOCATION "Resources"
    )

    qt_add_executable( Sigma MACOSX_BUNDLE src/main.cpp ${app_icon_macos} )
#! [appicon_macOS]
else()
    qt_add_executable( Sigma src/main.cpp )
endif()

    ${CMAKE_CURRENT_SOURCE_DIR}/third_party/alglib/*.h
)

    ${ALGLIB_SOURCES}
    ${CMAKE_CURRENT_SOURCE_DIR}/third_party/exprtk/exprtk.hpp
)

    ${CMAKE_CURRENT_SOURCE_DIR}/third_party/alglib
    ${CMAKE_CURRENT_SOURCE_DIR}/third_party/Eigen
)

set_target_properties( ThirdParty PROPERTIES
    AUTOMOC OFF
    AUTOUIC OFF
    AUTORCC OFF
)

    Qt6::Quick
    Qt6::Graphs
    ThirdParty
)

#    MACOSX_BUNDLE_GUI_IDENTIFIER com.example.Sigma

    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}

    WIN32_EXECUTABLE TRUE
